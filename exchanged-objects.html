navigator.credentials.create({
  publicKey?: PublicKeyCredentialCreationOptions {
    challenge: ArrayBuffer | TypedArray | DataView = supplied by server: Uint8Array,
    rp: {
      id?: string = domain,
      name: string = domain,
    },
    rpId: string = domain, # Undocumented on MDN, but required by browsers
    user: {
      displayName: string = " ",
      id: ArrayBuffer | TypedArray | DataView = new Uint8Array([0x20]),
      name: string = " ",
    },
    pubKeyCredParams: [{
        alg: number (enum-style) = -7,
        type: string (enum-style) = "public-key",
      }, {
        alg: number (enum-style) = -257,
        type: string (enum-style) = "public-key",
    }],
    attestation?: string (enum-style) = "none",
    attestationFormats?: string (enum-style)[] = [],
    authenticatorSelection?: {
      authenticatorAttachment?: string (enum-stye),
      requireResidentKey?: bool = false, # Deprecated
      residentKey?: string (enum-style) = "discouraged",
      userVerification?: string (enum-style) = "preferred",
    },
    excludeCredentials?: [{
        id: ArrayBuffer | TypedArray | DataView,
        transports?: string (enum-style)[],
        type: string (enum-style),
    }] = [],
    extensions?: {},
    hints?: string (enum-style)[],
    timeout?: number, # ms
  },
  federated?: FederatedCredentialInit,
  password?: PasswordCredentialInit,
  signal?: AbortSignal,
}) -> Promise <FederatedCredential | PasswordCredential | PublicKeyCredential> {
  id: string = base64url(rawId),
  rawId: ArrayBuffer,
  response: AuthenticatorAttestationResponse {
    attestationObject: ArrayBuffer,
    clientDataJSON: ArrayBuffer,
  },
  type: string = "public-key",
  authenticatorAttachment: string (enum-style),
},

navigator.credentials.get({
  publicKey?: PublicKeyCredentialRequestOptions {
    challenge: ArrayBuffer | TypedArray | DataView = supplied by server: Uint8Array,
    rp: { # New way. Undocumented on MDN, but required by browsers
      id?: string = domain,
      name: string = domain,
    },
    rpId?: string = domain, # Old way, still required for production. Not actually optional
    allowCredentials?: [{
        id: ArrayBuffer | TypedArray | DataView = supplied by server,
        type: string (enum-style) = "public-key",
        transports?: string (enum style)[],
    }, ...],
    extensions?: {},
    hints?: string (enum-style)[],
    timeout?: number, # ms
    userVerification?: string (enum-style) = "preferred",
  },
  mediation?: string (enum-style) = "optional",
  signal?: AbortSignal,
  password?: bool,
  identity?: IdentityCredentialRequestOptions,
  federated?: {}, # Deprecated
  otp?: string (enum style)[],
}) -> Promise <FederatedCredential | IdentityCredential |
OTPCredential | PasswordCredential | PublicKeyCredential> {
  id: string = base64url(rawId),
  rawId: ArrayBuffer,
  response: AuthenticatorAssertionResponse {
    authenticatorData: ArrayBuffer,
    clientDataJSON: ArrayBuffer,
    signature: ArrayBuffer,
    userHandle: ArrayBuffer | null,
  },
  type: string = "public-key",
  authenticatorAttachment: string (enum-style),
},
